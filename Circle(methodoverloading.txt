class  MethodOverloading1{
    public static void main(String[] args) {
        Circle c=new Circle();
        System.out.println(c.area(20));
        System.out.println(c.area(20.5f));
    }
}
class Circle{
    double area(int r){
        return 3.14*r*r;
    }
    double area(float r){
        return 3.14*r*r;
    }
}
